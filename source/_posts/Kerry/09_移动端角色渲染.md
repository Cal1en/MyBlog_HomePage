# 导入动画
从角色主模型中创建泛型的骨骼，并设置角色动画使用该骨骼。

如果发现角色动画脚部没有完全贴地且出现滑步抖动，我们可以选择关闭动画的`重采样曲线`，或关闭动画的压缩即可。

# 分辨金属度贴图和粗糙度贴图
金属度和粗糙度一般会存储在同一张贴图的两个通道中，我们可以通过金属度贴图的对比度远大于粗糙度来分辨。

```csharp
//具体情况具体分析
fixed3 compMask = tex2D(_CompMask, i.uv);
half metal = compMask.g;
half roughness = compMask.r;
```

# 金属度的处理
金属物体表面只有镜面反射，而没有漫反射，而非金属表面两者都有，因此我们需要使用金属度来区分二者，即作为漫反射和镜面反射的比例系数：

```csharp
half3 diffuseColor = albedo * (1 - metal);
half3 specColor = lerp(0, albedo, metal);
```

+ `half3 diffuseColor = albedo * (1 - metal);`：显然金属物体的`metal = 1`，因此`diffuseColor = (0,0,0)`，以此类推；
+ `half3 specColor = lerp(0.04, albedo, metal);`：这是一个经验公式，表示非金属物体表面的反射率$F_0$大约为`4%`，此处我们直接使用`0`，因为我们只希望人物的皮肤表面具有一定的高光，我们会在后续的直接光镜面反射部分再处理。

# 粗糙度的处理
粗糙度用于描述物体表面次反射的情况，物体表面越粗糙，镜面反射就越弱，因此我们可以使用粗糙度参与`Blinn-Phong`镜面反射的计算：

```csharp
//Roughness
half roughness = compMask.r;
    //PBR直转Blinn-Phong：half smoothness = RoughnessToSpecPower(roughness);
half smoothness = 1 - roughness;
smoothness = lerp(1, _SpecShiness, smoothness) * smoothness;    //再次乘以smoothness
                                                                //得到指数变化
```

# 直接光
## 直接光漫反射
使用`Lambert`光照模型：

```csharp
//Direct Diffuse
half3 directDiffuse = _LightColor0.rgb * diffuseColor * saturate(dot(lightDir, bumpWorldNormal));
directDiffuse *= atten;
```

## 直接光镜面反射
使用`Blinn-Phong`光照模型，同时还需要用到上述的粗糙度：

```csharp
//Direct Specular
//Direct Specular
half3 halfDir = normalize(viewDir + lightDir);
//specColor = lerp(specColor, 0.1, skinArea);
half3 directSpec = _LightColor0.rgb * specColor * pow(saturate(dot(halfDir, bumpWorldNormal)), smoothness);
directSpec *= atten;
```

+ `specColor = lerp(specColor, 0.1, skinArea);`：保证皮肤区域具有一定的高光反射。

# 间接光
## 间接光漫反射
我们需要使用到球谐函数，但是由于球谐函数的结果和当前使用的天空球有关，因此我们可以自己定义一个函数来计算球谐函数的结果，同时从我们所需要的`CubeMap`中提取出球谐函数参量;

同时作为漫反射，我们还需要添加菲涅尔因子，但是由于间接光理论上不会出现死黑，因此我们将使用 **`Half Lambert`** 光照模型：

```csharp
//Custom SH
half4 _Custom_SHAr, _Custom_SHAg, _Custom_SHAb;
half4 _Custom_SHBr, _Custom_SHBg, _Custom_SHBb;
half4 _Custom_SHC;

half3 CustomSH9(half3 normal)
{
    //ShadeSH9
    half3 res;
    //L0 + L1
    res.r = dot(_Custom_SHAr, float4(normal.xyz, 1.0));
    res.g = dot(_Custom_SHAg, float4(normal.xyz, 1.0));
    res.b = dot(_Custom_SHAb, float4(normal.xyz, 1.0));

    //L2的前四项
    half4 n2 = normal.xyzz * normal.yzzx;
    res.r += dot(_Custom_SHBr, n2);
    res.g += dot(_Custom_SHBg, n2);
    res.b += dot(_Custom_SHBb, n2);

    //L2的最后项
    res += _Custom_SHC.rgb * (normal.x * normal.x - normal.y * normal.y);

    return res;
}

//frag
    //Indirect Diffuse
    half3 halfLambert = saturate((dot(lightDir, bumpWorldNormal) + 1) * 0.5);
    half3 indirectDiffuse = CustomSH9(bumpWorldNormal) * diffuseColor * halfLambertTerm;
    indirectDiffuse = lerp(indirectDiffuse * 0.5, indirectDiffuse * _IndirectDiff, skinArea);
```

+ `indirectDiffuse = lerp(indirectDiffuse * 0.5, indirectDiffuse * _IndirectDiff, skinArea);`：单独调整皮肤区域的亮度。

## 间接光镜面反射
使用IBL：

```csharp
//Indirect Specular
roughness = roughness * (1.7 - 0.7 * roughness);    //缓入缓出
half mipLevel = roughness * 6;
half3 IBLColor = DecodeHDR(texCUBElod(_EnvMap, float4(reflectDir, mipLevel)), _EnvMap_HDR);
half3 indirectSpec = IBLColor * specColor * _Expose * halfLambert;
```

# 添加ACES
```csharp
Shader "CS04/Lethe_Body"
{
    Properties
    {
        _MainTex ("Texture", 2D) = "white" {}
        _CompMask("CompMask(RM)", 2D) = ""{}
        _BumpMap("BumpMap", 2D) = ""{}

        [Header(Adjust)]
        _SpecShiness("Spec Shiness", Range(1,10)) = 5
        _RoughnessAdjust("Roughness Adjust", Range(-1,1)) = 0
        _MetalAdjust("Metal Adjust", Range(-1,1)) = 0

        [Header(Indirect Spec)]
        _EnvMap("EnvMap", Cube) = ""{}
        _Tint("Tint", Color) = (1,1,1,1)
        _Rotate("Rotate", Range(0,360)) = 0
        _Expose("Expose", Range(1, 5)) = 1


        [HideInInspector]_Custom_SHAr("Custom_SHAr", Vector) = (0,0,0,0)
        [HideInInspector]_Custom_SHAg("Custom_SHAg", Vector) = (0,0,0,0)
        [HideInInspector]_Custom_SHAb("Custom_SHAb", Vector) = (0,0,0,0)
        [HideInInspector]_Custom_SHBr("Custom_SHBr", Vector) = (0,0,0,0)
        [HideInInspector]_Custom_SHBg("Custom_SHBg", Vector) = (0,0,0,0)
        [HideInInspector]_Custom_SHBb("Custom_SHBb", Vector) = (0,0,0,0)
        [HideInInspector]_Custom_SHC("Custom_SHC", Vector) = (0,0,0,0)
    }
    SubShader
    {
        Tags { "RenderType"="Opaque" }
        LOD 100

        Pass
        {
            Tags { "LightMode" = "ForwardBase" }
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_fwdbase

            #include "UnityCG.cginc"
            #include "Lighting.cginc"
            #include "AutoLight.cginc"

            struct v2f
            {
                float2 uv : TEXCOORD0;
                float4 pos : SV_POSITION;
                float3 worldNormal : TEXCOORD1;
                float3 worldPos : TEXCOORD2;
                float3 worldBinormal : TEXCOORD3;
                float3 worldTangent : TEXCOORD4;
                SHADOW_COORDS(5)
            };

            sampler2D _MainTex;
            float4 _MainTex_ST;
            sampler2D _CompMask;
            sampler2D _BumpMap;
            float _SpecShiness;

            fixed _RoughnessAdjust;
            fixed _MetalAdjust;

            samplerCUBE _EnvMap;
            float4 _EnvMap_HDR;
            half4 _Tint;
            float _Rotate;
            half _Expose;

            //Custom SH
            half4 _Custom_SHAr, _Custom_SHAg, _Custom_SHAb;
            half4 _Custom_SHBr, _Custom_SHBg, _Custom_SHBb;
            half4 _Custom_SHC;

            half3 CustomSH9(half3 normal)
            {
                //ShadeSH9
                half3 res;
                //L0 + L1
                res.r = dot(_Custom_SHAr, float4(normal.xyz, 1.0));
                res.g = dot(_Custom_SHAg, float4(normal.xyz, 1.0));
                res.b = dot(_Custom_SHAb, float4(normal.xyz, 1.0));

                //L2的前四项
                half4 n2 = normal.xyzz * normal.yzzx;
                res.r += dot(_Custom_SHBr, n2);
                res.g += dot(_Custom_SHBg, n2);
                res.b += dot(_Custom_SHBb, n2);

                //L2的最后项
                res += _Custom_SHC.rgb * (normal.x * normal.x - normal.y * normal.y);

                return res;
            }

            inline float3 ACESFilm(float3 x)
            {
                float a = 2.51f;
                float b = 0.03f;
                float c = 2.43f;
                float d = 0.59f;
                float e = 0.14f;
    
                return saturate((x*(a*x+b))/(x*(c*x+d)+e));
            }

            v2f vert (appdata_tan v)
            {
                v2f o;
                o.pos = UnityObjectToClipPos(v.vertex);
                o.uv = TRANSFORM_TEX(v.texcoord, _MainTex);
                o.worldNormal = UnityObjectToWorldNormal(v.normal);
                o.worldPos = UnityObjectToWorldDir(v.vertex);
                o.worldTangent = UnityObjectToWorldDir(v.tangent);
                o.worldBinormal = normalize(cross(o.worldNormal, o.worldTangent)) * v.tangent.w;
                TRANSFER_SHADOW(o)
                return o;
            }

            half4 frag (v2f i) : SV_Target
            {
                //Texture Info
                fixed3 albedo = tex2D(_MainTex, i.uv);
                fixed3 compMask = tex2D(_CompMask, i.uv);
                half3 bumpTangentNormal = UnpackNormal(tex2D(_BumpMap, i.uv));

                //Metal
                half metal = saturate(compMask.g + _MetalAdjust);
                half3 diffuseColor = albedo * (1 - metal);
                half3 specColor = lerp(0.04, albedo, metal);

                //Roughness
                half roughness = saturate(compMask.r + _RoughnessAdjust);
                    //half smoothness = RoughnessToSpecPower(roughness);
                half smoothness = 1 - roughness;
                smoothness = lerp(1, _SpecShiness, smoothness) * smoothness;   
                
                //TBN
                half3 worldTangent = normalize(i.worldTangent);
                half3 worldBinormal = normalize(i.worldBinormal);
                half3 worldNormal = normalize(i.worldNormal);

                //Bump
                half3 bumpWorldNormal = worldTangent * bumpTangentNormal.x + 
                                        worldBinormal * bumpTangentNormal.y + 
                                        worldNormal * bumpTangentNormal.z;

                //Light Info
                half3 viewDir = normalize(UnityWorldSpaceViewDir(i.worldPos));
                half3 lightDir = normalize(UnityWorldSpaceLightDir(i.worldPos));
                half3 reflectDir = normalize(reflect(-viewDir, bumpWorldNormal));
                UNITY_LIGHT_ATTENUATION(atten, i, i.worldPos)

                //Direct Diffuse
                half3 directDiffuse = _LightColor0.rgb * diffuseColor * saturate(dot(lightDir, bumpWorldNormal));
                directDiffuse *= atten;
                
                //Direct Specular
                half3 halfDir = normalize(viewDir + lightDir);
                half3 directSpec = _LightColor0.rgb * specColor * pow(saturate(dot(halfDir, bumpWorldNormal)), smoothness);
                directSpec *= atten;

                //Indirect Diffuse
                half3 halfLambert = saturate((dot(lightDir, bumpWorldNormal) + 1) * 0.5);
                half3 indirectDiffuse = CustomSH9(bumpWorldNormal) * diffuseColor * halfLambert;

                //IBL Info
                half s,c;
                sincos(radians(_Rotate), s, c);
                reflectDir.xz = mul(half2x2(c, -s, s, c), reflectDir.xz);
                reflectDir = normalize(reflectDir);

                //Indirect Specular
                roughness = roughness * (1.7 - 0.7 * roughness);    //缓入缓出
                half mipLevel = roughness * 6;
                half3 IBLColor = DecodeHDR(texCUBElod(_EnvMap, float4(reflectDir, mipLevel)), _EnvMap_HDR);
                half3 indirectSpec = IBLColor * specColor * _Expose * halfLambert;

                half3 finalColor = directDiffuse + directSpec + indirectDiffuse * 0.5 + indirectSpec;
                finalColor = ACESFilm(finalColor);
                return half4(finalColor, 1);
            }
            ENDCG
        }
    }
    Fallback "Diffuse"
}

```

# SSS皮肤
SSS指的是次表面散射，为了模拟这种效果，我们可以通过采样一张渐变纹理的方式来得到。

一般情况下我们会使用曲率图对渐变纹理进行采样，但是当模型面数较少时，我们可以直接使用$\vec{n} \cdot \vec{l}$进行采样。

<!-- 这是一张图片，ocr 内容为： -->
![](/img/Kerry/09_移动端角色渲染/1776824822497-64f7b34b-3481-47db-8401-7c0f9110b197.png)

我们可以在金属度贴图中分离出皮肤的区域，将其存入`CompMask`的`B`通道中。

```csharp
//Direct Diffuse
half diffuseTerm = saturate(dot(lightDir, bumpWorldNormal));
half3 directDiffuse = _LightColor0.rgb * diffuseColor * diffuseTerm;
directDiffuse *= atten;

//SSS Skin
half3 halfLambertTerm = saturate((diffuseTerm + 1) * 0.5);
half skinArea = 1 - compMask.b;
half2 uv_LUT = half2(diffuseTerm * atten + _LUTOffset, _CurveOffset);
half3 skinLUTColor = tex2D(_SkinLUT, uv_LUT);
half3 sssDiffuse = _LightColor0.rgb * diffuseColor * skinLUTColor * halfLambertTerm;
//sssDiffuse *= atten;

//SSS + Direct Diffuse
directDiffuse = lerp(directDiffuse, sssDiffuse, skinArea);
```

此处我们选择直接将`uv_LUT`的采样点乘以`atten`来模拟阴影/衰减效果，同时使用`skinArea`作为普通的漫反射与SSS漫反射之间的插值因子。

# 头发
## 各向异性渲染
一颗光滑的球被灯光照到，默认高光函数会形成一个圆斑，因为我们使用$\vec{n} \cdot \vec{h}$来计算高光，而光滑的球面的法线变化率在任意方向上都是相同的，这被称为**各向同性**。而如果一个表面在垂直方向和水平方向上的法线或其他用于计算光照的属性的变化率是不同的，则有可能形成条纹，这被称为**各向异性**光照。

由上图的各向异性效果可知，我们可以使用各向异性渲染来模拟头发丝的效果，显然我们有两种做法来制作各向异性渲染，一是使用法线贴图，在绘制时保证其在水平和垂直方向上的变化率不同，但是由于头发丝过于密集，且受制于贴图分辨率，我们一般不使用这种方法；二是使用算法进行模拟。

## Kajiya-Kay 光照模型
<!-- 这是一张图片，ocr 内容为： -->
![](/img/Kerry/09_移动端角色渲染/1776849505656-5c23fad2-542b-4dce-8e6f-a81049ece148.png)

`Kajiya-Kay`算法简称`kk`，用于模拟头发的高光。如果我们直接使用`Blinn-Phong`的高光，对单根头发来说，其沿着发丝方向法线的变化率肯定远小于沿着横截面外围的变化率，因此会出现大量沿着发丝方向的高光；对于多根头发组成的面片来说，其在上下左右两个方向上的法线变化率不大，因此会出现大面积的高光。这两种效果都不是我们想要的，**我们希望头发的高光应是类似于“天使环”的效果，围绕着头部形成一个圆环**，因此我们需要使用`Kajiya-Kay`算法，其原理如下：

我们原先使用半角向量和法线向量之间的夹角参与高光计算，但是法线方向的变化率在发丝方向上变化率小且平滑，在横截面方向上突变，我们希望这两种变化方向能够反过来，**也就是我们希望找到一个在横截面方向上变化率小，而在发丝方向上变化率大的向量，也就是当前顶点的副切线方向。** 即上图中的向量$\vec{T}$，其表示的是沿着发丝方向的切线向量。

因此，我们可以使用副切线方向和半角向量之间的夹角参与高光计算，但是原先使用法线向量和半角向量之间的夹角时，高光强度随着夹角的减小而增大，但是显然当我们使用副切线方向时，高光强度会在副切线垂直于半角向量时达到最大，因此我们需要使用$sin(\vec{T}, \vec{H})$进行计算。但是直接使用`Unity Shader`中的`sin`函数会浪费大量计算资源，因此我们可以使用$\sqrt{1 - cos(\vec{T}, \vec{H})^2}$进行计算，其中$cos(\vec{T}, \vec{H})$可以使用`dot()`获得。

```csharp
half3 viewDir = normalize(UnityWorldSpaceViewDir(i.worldPos));
half3 lightDir = normalize(UnityWorldSpaceLightDir(i.worldPos));
half3 halfDir = normalize(viewDir + lightDir);
half3 worldNormal = normalize(i.worldNormal);
half3 worldBinormal = normalize(i.worldBinormal);

//调整高光位置
worldBinormal = normalize(worldBinormal + worldNormal * _ShiftOffset);

fixed4 albedo = tex2D(_MainTex, i.uv);
half3 specTerm = dot(halfDir, worldBinormal);
half3 specColor = _LightColor0.rgb * albedo.rgb * pow(sqrt(1 - specTerm * specTerm), _Gloss);

return half4(specColor, 1);
```

## 高光扰动
接下来我们需要在`kk`的基础之上制作头发丝的效果，我们在上面使用了`worldBinormal = normalize(worldBinormal + worldNormal * _ShiftOffset);`来调整高光在`副切线*法线`平面上的位移，那么同理，我们可以让一些顶点向副切线方向偏移，一些顶点向法线方向偏移，就能做出高光扰动的效果。由于我们需要让发生偏移的顶点之间的距离很小才能模拟头发丝的效果，我们将引入扰动贴图。

```csharp
Shader "Unlit/CS04_KKAniso"
{
    Properties
    {
        _MainTex ("Texture", 2D) = "white" {}
        _Gloss("Gloss", Float) = 5
        _ShiftOffset("Shift Offset", Range(-1,1)) = 0
        _ShiftMap("Shift Map", 2D) = ""{}
        _NoiseIntensity("Noise Intensity", Range(-2,2)) = 1
    }
    SubShader
    {
        Tags { "RenderType"="Opaque" }
        LOD 100

        Pass
        {
            Tags {"LightMode"="ForwardBase"}

            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag

            #include "UnityCG.cginc"
            #include "Lighting.cginc"

            struct v2f
            {

                //xy MainTex; zw ShiftMap
                float4 uv : TEXCOORD0;
                float4 pos : SV_POSITION;
                float3 worldPos : TEXCOORD1;
                float3 worldNormal : TEXCOORD2;
                float3 worldBinormal : TEXCOORD3;
            };

            sampler2D _MainTex;
            float4 _MainTex_ST;
            float _Gloss;
            fixed _ShiftOffset;
            sampler2D _ShiftMap;
            float4 _ShiftMap_ST;
            half _NoiseIntensity;

            v2f vert (appdata_tan v)
            {
                v2f o;
                o.pos = UnityObjectToClipPos(v.vertex);
                o.uv.xy = TRANSFORM_TEX(v.texcoord, _MainTex);
                o.uv.zw = TRANSFORM_TEX(v.texcoord, _ShiftMap);
                o.worldPos = mul(unity_ObjectToWorld, v.vertex);
                o.worldNormal = UnityObjectToWorldNormal(v.normal);
                o.worldBinormal = normalize(cross(o.worldNormal, UnityObjectToWorldDir(v.tangent)) * v.tangent.w);
                return o;
            }

            half4 frag (v2f i) : SV_Target
            {
                half3 viewDir = normalize(UnityWorldSpaceViewDir(i.worldPos));
                half3 lightDir = normalize(UnityWorldSpaceLightDir(i.worldPos));
                half3 halfDir = normalize(viewDir + lightDir);
                half3 worldNormal = normalize(i.worldNormal);
                half3 worldBinormal = normalize(i.worldBinormal);
                
                //添加高光扰动
                half shiftNoise = tex2D(_ShiftMap, i.uv.zw);
                shiftNoise = (shiftNoise * 2 - 1) * _NoiseIntensity;
                half3 shiftOffset = worldNormal * (_ShiftOffset + shiftNoise);
                //调整高光位置
                worldBinormal = normalize(worldBinormal + shiftOffset);
 
                fixed4 albedo = tex2D(_MainTex, i.uv.xy);
                half3 specTerm = dot(halfDir, worldBinormal);
                half3 specColor = _LightColor0.rgb * albedo.rgb * pow(sqrt(1 - specTerm * specTerm), _Gloss);

                return half4(specColor, 1);
            }
            ENDCG
        }
    }
}

```

## 改进Kajiya-Kay光照模型
```csharp
//Direct Specular
half3 halfDir = normalize(viewDir + lightDir);
half TdotH = dot(halfDir, worldTangent);
half NdotH = dot(bumpWorldNormal, halfDir);
half NdotV = saturate(dot(viewDir, bumpWorldNormal));
half aniso_atten = saturate(sqrt(saturate(halfLambertTerm / NdotV))) * atten;

//spec 1
half3 specColor_1 = _SpecColor_1 + specColor;
half3 anisoOffset_1 = bumpWorldNormal * (anisoNoise * _SpecNoise_1 + _SpecOffset_1);
half3 worldBinormal_1 = normalize(worldBinormal + anisoOffset_1);
half BdotH_1 = dot(halfDir, worldBinormal_1) / _SpecShiness_1;

half3 specTerm_1 = exp(-(TdotH * TdotH + BdotH_1 * BdotH_1) / (1 + NdotH));

//spec 2
half3 specColor_2 = _SpecColor_2 + specColor;
half3 anisoOffset_2 = bumpWorldNormal * (anisoNoise * _SpecNoise_2 + _SpecOffset_2);
half3 worldBinormal_2 = normalize(worldBinormal + anisoOffset_2);
half BdotH_2 = dot(halfDir, worldBinormal_2) / _SpecShiness_2;

half3 specTerm_2 = exp(-(TdotH * TdotH + BdotH_2 * BdotH_2) / (2 + NdotH));
```

+ `half aniso_atten = saturate(sqrt(saturate(halfLambertTerm / NdotV))) * atten;`：原先我们制作边缘光使用的是`1 - NdotV`，但是这种衰减是线性的，并没有指数衰减的效果好。因此我们引入`halfLambertTerm / NdotV`制作**指数衰减**的边缘光照。
+ `half3 specTerm_1 = exp(-(TdotH * TdotH + BdotH_1 * BdotH_1) / (1 + NdotH));`：我们使用**正态分布**函数作为高光的分布函数，将`_SpecShiness_1`作为标准差。

## 代码
头发需要关闭漫反射的衰减，以及去除间接光的漫反射。

```csharp
Shader "CS04/Lethe_Hair"
{
    Properties
    {
        [NoScaleOffset]_MainTex ("Texture", 2D) = "white" {}
        _BaseColor("Base Color", Color) = (1,1,1,1)
        [NoScaleOffset][Normal]_BumpMap("BumpMap", 2D) = ""{}

        [Header(Specular)][Space]
        _AnisoMap("Aniso Map", 2D) = "gray"{}

        _SpecShiness_1("Spec Shiness 1", Range(0,1)) = 0.1
        _SpecColor_1("Specular Color 1", Color) = (1,1,1,1)
        _SpecNoise_1("Specular Noise 1", Range(-2,2)) = 1
        _SpecOffset_1("Specular Offset 1", Range(-1,1)) = 0

        _SpecShiness_2("Spec Shiness 2", Range(0,1)) = 0.1
        _SpecColor_2("Specular Color 2", Color) = (1,1,1,1)
        _SpecNoise_2("Specular Noise 2", Range(-2,2)) = 1
        _SpecOffset_2("Specular Offset 2", Range(-1,1)) = 0

        [Header(Adjust)][Space]
        _RoughnessAdjust("Roughness Adjust", Range(-1,1)) = 0

        [Header(IBL)][Space]
        [NoScaleOffset]_EnvMap("EnvMap", Cube) = ""{}
        _Rotate("Rotate", Range(0,360)) = 0
        _Expose("Expose", Range(0, 5)) = 1

        [Header(Debug)][Space]
        [Toggle(_DIRECTDIFFUSE_ON)] _DirectDiffuseOn("Direct Diffuse On", Float) = 1
        [Toggle(_DIRECTSPEC_ON)] _DirectSpecOn("Direct Specular On", Float) = 1
        [Toggle(_INDIRECTSPEC_ON)] _IndirectSpecOn("Indirect Specular On", Float) = 1
    }
    SubShader
    {
        Tags { "RenderType"="Opaque" }
        LOD 100

        Pass
        {
            Tags { "LightMode" = "ForwardBase" }
            CGPROGRAM
            #pragma vertex vert
            #pragma fragment frag
            #pragma multi_compile_fwdbase

            #pragma shader_feature _DIRECTDIFFUSE_ON
            #pragma shader_feature _DIRECTSPEC_ON
            #pragma shader_feature _INDIRECTSPEC_ON

            #include "UnityCG.cginc"
            #include "Lighting.cginc"
            #include "AutoLight.cginc"

            struct v2f
            {
                //xy Diffuse; zw Aniso
                float4 uv : TEXCOORD0;
                float4 pos : SV_POSITION;
                float3 worldNormal : TEXCOORD1;
                float3 worldPos : TEXCOORD2;
                float3 worldBinormal : TEXCOORD3;
                float3 worldTangent : TEXCOORD4;
                SHADOW_COORDS(5)
            };

            sampler2D _MainTex;
            float4 _MainTex_ST;
            sampler2D _BumpMap;
            fixed4 _BaseColor;

            //Specular
            sampler2D _AnisoMap;
            float4 _AnisoMap_ST;

            fixed _SpecShiness_1;
            fixed4 _SpecColor_1;
            half _SpecNoise_1;
            fixed _SpecOffset_1;

            fixed _SpecShiness_2;
            fixed4 _SpecColor_2;
            half _SpecNoise_2;
            fixed _SpecOffset_2;

            //Adjust
            fixed _RoughnessAdjust;

            //IBL
            samplerCUBE _EnvMap;
            float4 _EnvMap_HDR;
            float _Rotate;
            half _Expose;

            inline float3 ACESFilm(float3 x)
            {
                float a = 2.51f;
                float b = 0.03f;
                float c = 2.43f;
                float d = 0.59f;
                float e = 0.14f;
    
                return saturate((x*(a*x+b))/(x*(c*x+d)+e));
            }

            v2f vert (appdata_tan v)
            {
                v2f o;
                o.pos = UnityObjectToClipPos(v.vertex);
                o.uv.xy = TRANSFORM_TEX(v.texcoord, _MainTex);
                o.uv.zw = TRANSFORM_TEX(v.texcoord, _AnisoMap);
                o.worldNormal = UnityObjectToWorldNormal(v.normal);
                o.worldPos = mul(unity_ObjectToWorld, v.vertex);
                o.worldTangent = UnityObjectToWorldDir(v.tangent);
                o.worldBinormal = normalize(cross(o.worldNormal, o.worldTangent)) * v.tangent.w;
                TRANSFER_SHADOW(o)
                return o;
            }

            half4 frag (v2f i) : SV_Target
            {
                //Texture Info
                fixed3 albedo = tex2D(_MainTex, i.uv.xy) * _BaseColor;
                half3 bumpTangentNormal = UnpackNormal(tex2D(_BumpMap, i.uv.xy));
                half anisoNoise = tex2D(_AnisoMap, i.uv.zw).r - 0.5;
                //half anisoNoise = tex2D(_AnisoMap, i.uv.zw).r * 2 - 1;

                half3 diffuseColor = albedo;
                half3 specColor = albedo;
                
                //TBN
                half3 worldTangent = normalize(i.worldTangent);
                half3 worldBinormal = normalize(i.worldBinormal);
                half3 worldNormal = normalize(i.worldNormal);

                //Bump
                half3 bumpWorldNormal = worldTangent * bumpTangentNormal.x + 
                                        worldBinormal * bumpTangentNormal.y + 
                                        worldNormal * bumpTangentNormal.z;

                //Light Info
                half3 viewDir = normalize(UnityWorldSpaceViewDir(i.worldPos));
                half3 lightDir = normalize(UnityWorldSpaceLightDir(i.worldPos));
                half3 reflectDir = normalize(reflect(-viewDir, bumpWorldNormal));
                UNITY_LIGHT_ATTENUATION(atten, i, i.worldPos)

                //Direct Diffuse
                half diffuseTerm = saturate(dot(lightDir, bumpWorldNormal));
                half3 halfLambertTerm = saturate((diffuseTerm + 1) * 0.5);

                #if defined(_DIRECTDIFFUSE_ON)
                    half3 directDiffuse = diffuseColor;
                #else
                    half3 directDiffuse = half3(0,0,0);
                #endif

                //Roughness
                half roughness = saturate(_RoughnessAdjust);

                //Direct Specular
                half3 halfDir = normalize(viewDir + lightDir);
                half TdotH = dot(halfDir, worldTangent);
                half NdotH = dot(bumpWorldNormal, halfDir);
                half NdotV = saturate(dot(viewDir, bumpWorldNormal));
                half aniso_atten = saturate(sqrt(saturate(halfLambertTerm / NdotV))) * atten;

                //spec 1
                half3 specColor_1 = _SpecColor_1 + specColor;
                half3 anisoOffset_1 = bumpWorldNormal * (anisoNoise * _SpecNoise_1 + _SpecOffset_1);
                half3 worldBinormal_1 = normalize(worldBinormal + anisoOffset_1);
                half BdotH_1 = dot(halfDir, worldBinormal_1) / _SpecShiness_1;

                half3 specTerm_1 = exp(-(TdotH * TdotH + BdotH_1 * BdotH_1) / (1 + NdotH));

                //spec 2
                half3 specColor_2 = _SpecColor_2 + specColor;
                half3 anisoOffset_2 = bumpWorldNormal * (anisoNoise * _SpecNoise_2 + _SpecOffset_2);
                half3 worldBinormal_2 = normalize(worldBinormal + anisoOffset_2);
                half BdotH_2 = dot(halfDir, worldBinormal_2) / _SpecShiness_2;

                half3 specTerm_2 = exp(-(TdotH * TdotH + BdotH_2 * BdotH_2) / (2 + NdotH));

                #if defined(_DIRECTSPEC_ON)
                    half3 directSpec_1 = _LightColor0.rgb * specColor_1 * specTerm_1;
                    half3 directSpec_2 = _LightColor0.rgb * specColor_2 * specTerm_2;
                    half3 directSpec = directSpec_1 + directSpec_2;
                    directSpec *= aniso_atten;
                #else
                    half3 directSpec = half3(0,0,0);
                #endif

                //IBL Info
                half s,c;
                sincos(radians(_Rotate), s, c);
                reflectDir.xz = mul(half2x2(c, -s, s, c), reflectDir.xz);
                reflectDir = normalize(reflectDir);

                //Indirect Specular
                roughness = roughness * (1.7 - 0.7 * roughness);    //缓入缓出
                half mipLevel = roughness * 6;
                half3 IBLColor = DecodeHDR(texCUBElod(_EnvMap, float4(reflectDir, mipLevel)), _EnvMap_HDR);
                #if defined(_INDIRECTSPEC_ON)
                    half3 indirectSpec = IBLColor * anisoNoise * _Expose * halfLambertTerm;
                #else
                    half3 indirectSpec = half3(0,0,0);
                #endif

                half3 finalColor = directDiffuse * 2 + directSpec + indirectSpec;
                
                finalColor = ACESFilm(finalColor);
                return half4(finalColor, 1);
            }
            ENDCG
        }
    }
    Fallback "Diffuse"
}

```
